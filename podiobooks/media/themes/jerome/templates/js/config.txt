function siteVars(opt){
	switch (opt){
		case 'home':
			return '{% url podiobooks.main.views.index %}';
			break;
		case 'theme':
			return '{{ settings.THEME_MEDIA_URL }}';
			break;
		case 'img':
			return siteVars('theme') + 'images/';
			break; 
		case 'css':
			return siteVars('theme') + 'css/';
			break; 
		case 'swf':
			return siteVars('theme') + 'flash/';
			break;
		case 'js':
			return siteVars('theme') + 'js/';
			break;
		default:
			return null;
			break;
	}
}
